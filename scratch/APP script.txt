const START_ROW_DATA = 2; 

// KOLOM KHUSUS UNTUK BACKUP ID TIDB & RAW JSON
const TIDB_ID_COL = 100;   // Kolom CV
const TIDB_JSON_COL = 101; // Kolom CW

function getDbIdByLine(userLine) {
  const str = String(userLine || "2").toUpperCase().trim(); 
  if (str.includes("LINE 1") || str.includes("LINE1") || str === "1") return ID_DB_LINE_1;
  if (str.includes("LINE 3") || str.includes("LINE3") || str === "3") return ID_DB_LINE_3;
  if (str.includes("LINE 4") || str.includes("LINE4") || str === "4") return ID_DB_LINE_4;
  return ID_DB_LINE_2; 
}

function raw(val) {
  if (val === undefined || val === null) return ""; 
  return val; 
}

function getRowByTidbId(sheet, tidbId, data, userLine) {
  if (!tidbId || tidbId === "saved") {
    tidbId = null;
  } else {
    var lastRow = sheet.getLastRow();
    if (lastRow < START_ROW_DATA) return -1;
    var numRows = lastRow - START_ROW_DATA + 1;
    var rangeCV = sheet.getRange(START_ROW_DATA, TIDB_ID_COL, numRows, 1).getValues();
    var rangeA = sheet.getRange(START_ROW_DATA, 1, numRows, 1).getValues();
    for (var i = numRows - 1; i >= 0; i--) {
      if (String(rangeCV[i][0]).trim() === String(tidbId).trim() || String(rangeA[i][0]).trim() === String(tidbId).trim()) {
        return i + START_ROW_DATA;
      }
    }
  }

  // Fallback pencarian berdasarkan keys unik: no_batch, tanggal, dan shift
  if (!data || !data.no_batch || !data.tanggal) return -1;
  var lastRow = sheet.getLastRow();
  if (lastRow < START_ROW_DATA) return -1;
  var numRows = lastRow - START_ROW_DATA + 1;
  
  var colBatch = 3;
  var colTgl = (String(userLine) === "3" || String(userLine) === "1") ? 4 : 5;
  var colShift = (String(userLine) === "3" || String(userLine) === "1") ? 5 : 6;
  
  var maxCol = Math.max(colBatch, colTgl, colShift);
  var values = sheet.getRange(START_ROW_DATA, 1, numRows, maxCol).getValues();
  
  var targetBatch = String(raw(data.no_batch)).trim().toLowerCase();
  var targetTgl = String(formatTglIndo(data.tanggal)).replace(/^'/, "").trim();
  var targetShift = String(raw(data.shift)).trim().toLowerCase();
  
  for (var i = numRows - 1; i >= 0; i--) {
    var rowBatch = String(values[i][colBatch - 1] || "").trim().toLowerCase();
    var rowTgl = String(values[i][colTgl - 1] || "").replace(/^'/, "").trim();
    var rowShift = String(values[i][colShift - 1] || "").trim().toLowerCase();
    
    if (rowBatch === targetBatch && rowTgl === targetTgl && rowShift === targetShift) {
      return i + START_ROW_DATA;
    }
  }
  return -1;
}

function getLastRowInColumn(sheet, columnLetter) { 
  const range = sheet.getRange(columnLetter + "1:" + columnLetter).getValues(); 
  for (let i = range.length - 1; i >= 0; i--) { if (range[i][0] !== "") return i + 1; } 
  return 1; 
}

function responseJSON(data) { 
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); 
}

function formatTglIndo(dateStr) {
  if (!dateStr) return "";
  var d = new Date(dateStr);
  if (isNaN(d.getTime())) return "'" + dateStr; 
  var day = ("0" + d.getDate()).slice(-2);
  var month = ("0" + (d.getMonth() + 1)).slice(-2);
  var year = d.getFullYear();
  return "'" + year + "-" + month + "-" + day; 
}

/**
 * Mencari baris paling akhir yang berisi data nyata (menghindari lompat ke bawah akibat format/border kosong).
 */
function getRealLastDataRow(sheet, minRow) {
  if (!sheet) return minRow;
  var lastRow = sheet.getLastRow();
  if (lastRow < minRow) return minRow;
  
  var values = sheet.getRange(1, 1, lastRow, 3).getValues();
  for (var i = values.length - 1; i >= minRow - 1; i--) {
    var colA = String(values[i][0] || "").trim();
    var colB = String(values[i][1] || "").trim();
    var colC = String(values[i][2] || "").trim();
    if (colA !== "" || colB !== "" || colC !== "") {
      return i + 2;
    }
  }
  return minRow;
}

function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);
  try {
    const params = JSON.parse(e.postData.contents);
    const action = params.action || "";
    const userLine = params.user ? (params.user.line || params.user.plant || "2") : "2"; 
    let result = {};

    switch (action) {
      case "register": result = handleRegister(params.data); break;
      case "verify": result = handleVerification(params.data); break;
      case "login": result = handleLogin(params.data); break;
      case "update_password": result = handleUpdatePassword(params.data); break;
      case "get_validation_data": result = getValidationData(); break;
      case "update_foto": result = handleUpdateFoto(params.data); break;
      case "send_feedback": result = handleSendFeedback(params.data, params.user); break;

      case "get_pending_approvals": result = getPendingApprovals(); break;
      case "approve_user": result = approveUser(params.data); break;
      case "check_approval_status": result = checkApprovalStatus(params.data); break;
      case "get_onesheet_data": result = getOnesheetData(params.data, params.user, userLine); break;
      
      case "submit_reject_c": result = saveRejectZoneC(params.data, params.user, userLine); break; 
      case "submit_reject_f": result = saveRejectZoneF(params.data, params.user, userLine); break; 
      case "submit_downtime_c": result = saveDowntimeZoneC(params.data, params.user, userLine); break;
      case "submit_downtime_f": result = saveDowntimeZoneF(params.data, params.user, userLine); break;
      case "update_reject_c": result = updateRejectZoneC(params.data, params.user, userLine); break;
      case "delete_reject_c": result = deleteRejectZoneC(params.data, params.user, userLine); break;
      case "update_reject_f": result = updateRejectZoneF(params.data, params.user, userLine); break;
      case "delete_reject_f": result = deleteRejectZoneF(params.data, params.user, userLine); break;
      case "update_downtime_c": result = updateDowntimeZoneC(params.data, params.user, userLine); break;
      case "delete_downtime_c": result = deleteDowntimeZoneC(params.data, params.user, userLine); break;
      case "update_downtime_f": result = updateDowntimeZoneF(params.data, params.user, userLine); break;
      case "delete_downtime_f": result = deleteDowntimeZoneF(params.data, params.user, userLine); break;

      case "get_today_reject_c": result = getTodayData(userLine, SHEET_DL_REJECT_C, 74, 3, 4); break;
      case "get_today_reject_f": result = getTodayData(userLine, SHEET_DL_REJECT_F, 67, 4, 4); break;
      case "get_today_downtime_c": result = getTodayData(userLine, SHEET_DL_DOWNTIME_C, 16, 2, 2); break;
      case "get_today_downtime_f": result = getTodayData(userLine, SHEET_DL_DOWNTIME_F, 17, 2, 2); break;

      default:
        result = { status: "error", message: "Action tidak dikenal: " + action };
    }
    return responseJSON(result);
  } catch (error) {
    if (e && e.postData && e.postData.contents) {
      try {
        const p = JSON.parse(e.postData.contents);
        if (p.action && (p.action.indexOf('submit_') !== -1 || p.action.indexOf('update_') !== -1)) {
          const uLine = p.user ? (p.user.line || p.user.plant || "2") : "2";
          queuePendingSave(uLine, p.action, p.data, p.user);
        }
      } catch (ex) {}
    }
    return responseJSON({ status: "error", message: error.toString(), queued: true });
  } finally {
    lock.releaseLock();
  }
}

function saveRejectZoneC(data, user, userLine) {
  const ss = SpreadsheetApp.openById(getDbIdByLine(userLine));
  const sheet = ss.getSheetByName(SHEET_DL_REJECT_C);
  if (!sheet) return { status: 'error', message: `Sheet Missing!` };
  var idToSave = data.original_id || data.id;
  if (idToSave) {
    var existingRow = getRowByTidbId(sheet, idToSave, data, userLine);
    if (existingRow !== -1) return updateRejectZoneC(data, user, userLine);
  }
  var startRow = getRealLastDataRow(sheet, 4);
  
  var rowData = constructRejectCArray(data, user, data.is_closing, startRow, userLine);
  sheet.getRange(startRow, 1, 1, rowData.length).setValues([rowData]);
  
  // Real-Time Sync Backup (ID TiDB & JSON Payload)
  sheet.getRange(startRow, TIDB_ID_COL).setValue(idToSave || "");
  sheet.getRange(startRow, TIDB_JSON_COL).setValue(JSON.stringify(data));

  if (data.is_closing === true) sheet.getRange(startRow, 1, 1, rowData.length).setBackground("#FFF2CC");
  SpreadsheetApp.flush();
  return { status: 'success', message: 'Data Zone C Tersimpan!' };
}

function updateRejectZoneC(data, user, userLine) {
  const ss = SpreadsheetApp.openById(getDbIdByLine(userLine));
  const sheet = ss.getSheetByName(SHEET_DL_REJECT_C);
  if (!sheet) return { status: 'error', message: `Sheet Missing!` };
  
  // Gunakan ID TiDB untuk mencari baris yang tepat
  var idToUpdate = data.original_id || data.id;
  var rowIndex = getRowByTidbId(sheet, idToUpdate, data, userLine);
  if (rowIndex === -1) return saveRejectZoneC(data, user, userLine); 
  
  var rowData = constructRejectCArray(data, user, data.is_closing, rowIndex, userLine);
  rowData[0] = sheet.getRange(rowIndex, 1).getValue(); // Pertahankan Timestamp awal
  rowData[1] = (user && user.nama) ? user.nama : "Unknown (Edited)"; 
  
  sheet.getRange(rowIndex, 1, 1, rowData.length).setValues([rowData]);
  
  sheet.getRange(rowIndex, TIDB_ID_COL).setValue(idToUpdate || "");
  sheet.getRange(rowIndex, TIDB_JSON_COL).setValue(JSON.stringify(data));

  if (data.is_closing === true) sheet.getRange(rowIndex, 1, 1, rowData.length).setBackground("#FFF2CC");
  else sheet.getRange(rowIndex, 1, 1, rowData.length).setBackground(null); 
  SpreadsheetApp.flush();
  return { status: 'success', message: 'Data OEE Zone C Berhasil Diupdate!' };
}

function deleteRejectZoneC(data, user, userLine) {
  const ss = SpreadsheetApp.openById(getDbIdByLine(userLine));
  const sheet = ss.getSheetByName(SHEET_DL_REJECT_C);
  if (!sheet) return { status: 'error', message: `Sheet Missing!` };
  var idToDelete = data.original_id || data.id;
  var rowIndex = getRowByTidbId(sheet, idToDelete, data, userLine);
  if (rowIndex !== -1) {
    sheet.deleteRow(rowIndex);
  } else {
    queuePendingDelete(userLine, SHEET_DL_REJECT_C, idToDelete);
  }
  SpreadsheetApp.flush();
  return { status: 'success', message: 'Data berhasil dihapus!' };
}

function normalizeGasDate(val) {
  if (!val) return "";
  if (val.getTime) {
    var y = val.getFullYear();
    var m = ("0" + (val.getMonth() + 1)).slice(-2);
    var d = ("0" + val.getDate()).slice(-2);
    return y + "-" + m + "-" + d;
  }
  var str = String(val).replace(/'/g, "").trim();
  if (str.indexOf("/") !== -1) {
    var parts = str.split("/");
    if (parts.length === 3) {
      var day = parts[0].padStart(2, '0');
      var month = parts[1].padStart(2, '0');
      var year = parts[2];
      return year + "-" + month + "-" + day; 
    }
  }
  if (str.length >= 10) {
    return str.substring(0, 10);
  }
  return str;
}

// ==================== FUNGSI TRIGGER WAKTU 1 MENIT (TIME-DRIVEN TRIGGER) ====================
/**
 * FUNGSI TRIGGER AUTOMATIS 1 MENIT
 * Pasang fungsi ini pada Trigger Google Apps Script (Time-driven -> Every 1 minute).
 * 
 * Fungsi ini bertugas:
 * 1. Memeriksa dan mengeksekusi antrean penghapusan data (Queue Delete) yang tertunda saat lock sibuk.
 * 2. Membersihkan baris duplikat berdasarkan ID TiDB pada setiap Line.
 * 3. Memastikan sinkronisasi data yang telah di-delete dari sistem terupdate otomatis di Google Spreadsheet.
 */
function autoSyncAndClean1Min() {
  // 1. Proses Antrean Deletion & Saves
  processPendingDeletes();
  processPendingSaves();

  // 2. Pembersihan Duplikat ID TiDB pada Sheet OEE & Downtime
  const lines = ["1", "2", "3", "4"];
  const sheetNames = [SHEET_DL_REJECT_C, SHEET_DL_REJECT_F, SHEET_DL_DOWNTIME_C, SHEET_DL_DOWNTIME_F];
  
  for (let l = 0; l < lines.length; l++) {
    try {
      const ss = SpreadsheetApp.openById(getDbIdByLine(lines[l]));
      for (let s = 0; s < sheetNames.length; s++) {
        const sheet = ss.getSheetByName(sheetNames[s]);
        if (!sheet) continue;
        
        const lastRow = sheet.getLastRow();
        if (lastRow < START_ROW_DATA) continue;
        
        const numRows = lastRow - START_ROW_DATA + 1;
        const idsCV = sheet.getRange(START_ROW_DATA, TIDB_ID_COL, numRows, 1).getValues();
        const idsA = sheet.getRange(START_ROW_DATA, 1, numRows, 1).getValues();
        
        const seenIds = new Set();
        const rowsToDelete = [];
        
        // Cek dari bawah ke atas (data terbaru di bawah)
        for (let i = numRows - 1; i >= 0; i--) {
          let id = String(idsCV[i][0] || idsA[i][0] || "").trim();
          
          if (id && !isNaN(Number(id)) && Number(id) > 0) {
            if (seenIds.has(id)) {
              rowsToDelete.push(i + START_ROW_DATA);
            } else {
              seenIds.add(id);
            }
          }
        }
        
        // Hapus baris duplikat dari bawah agar indeks tidak bergeser
        rowsToDelete.sort((a, b) => b - a);
        for (let r = 0; r < rowsToDelete.length; r++) {
          sheet.deleteRow(rowsToDelete[r]);
        }
      }
    } catch (err) {
      console.error(`Error clean line ${lines[l]}: ${err.toString()}`);
    }
  }
}

function queuePendingDelete(userLine, sheetName, idToDelete) {
  if (!idToDelete || idToDelete === "saved") return;
  try {
    const props = PropertiesService.getScriptProperties();
    const queueStr = props.getProperty("PENDING_DELETES") || "[]";
    const queue = JSON.parse(queueStr);
    if (!queue.some(item => String(item.id) === String(idToDelete) && item.sheet === sheetName)) {
      queue.push({ line: userLine, sheet: sheetName, id: idToDelete, timestamp: new Date().getTime(), retries: 0 });
      props.setProperty("PENDING_DELETES", JSON.stringify(queue));
    }
  } catch (e) { console.error("Failed to queue delete:", e); }
}

function processPendingDeletes() {
  try {
    const props = PropertiesService.getScriptProperties();
    const queueStr = props.getProperty("PENDING_DELETES") || "[]";
    let queue = JSON.parse(queueStr);
    if (!queue || queue.length === 0) return;
    
    const remainingQueue = [];
    for (let i = 0; i < queue.length; i++) {
      const item = queue[i];
      if (new Date().getTime() - item.timestamp > 86400000) continue;
      
      const ss = SpreadsheetApp.openById(getDbIdByLine(item.line));
      const sheet = ss.getSheetByName(item.sheet);
      if (!sheet) continue;
      
      const rowIndex = getRowByTidbId(sheet, item.id);
      if (rowIndex !== -1) {
        sheet.deleteRow(rowIndex);
      } else {
        item.retries = (item.retries || 0) + 1;
        if (item.retries < 10) remainingQueue.push(item);
      }
    }
    props.setProperty("PENDING_DELETES", JSON.stringify(remainingQueue));
  } catch (e) { console.error("Error processing delete queue:", e); }
}

function queuePendingSave(userLine, action, data, user) {
  if (!data) return;
  try {
    const props = PropertiesService.getScriptProperties();
    const queueStr = props.getProperty("PENDING_SAVES") || "[]";
    const queue = JSON.parse(queueStr);
    var targetId = data.original_id || data.id || (data.no_batch + "_" + data.tanggal + "_" + data.shift);
    const filtered = queue.filter(item => {
      var itemId = item.data.original_id || item.data.id || (item.data.no_batch + "_" + item.data.tanggal + "_" + item.data.shift);
      return String(itemId) !== String(targetId) || item.action !== action;
    });
    filtered.push({ line: userLine, action: action, data: data, user: user, timestamp: new Date().getTime(), retries: 0 });
    props.setProperty("PENDING_SAVES", JSON.stringify(filtered));
  } catch (e) { console.error("Failed to queue save:", e); }
}

function processPendingSaves() {
  try {
    const props = PropertiesService.getScriptProperties();
    const queueStr = props.getProperty("PENDING_SAVES") || "[]";
    let queue = JSON.parse(queueStr);
    if (!queue || queue.length === 0) return;
    
    const remainingQueue = [];
    const now = new Date().getTime();
    for (let i = 0; i < queue.length; i++) {
      const item = queue[i];
      // Buang antrean yang lebih lama dari 1 jam (3600000 ms) atau data kadaluwarsa/stuck seperti ID 1690
      if (now - item.timestamp > 3600000) continue;
      if (item.data && (item.data.original_id == 1690 || item.data.id == 1690 || item.data.no_batch == "1D1A6G014")) continue;
      
      try {
        if (item.action === "submit_reject_c") saveRejectZoneC(item.data, item.user, item.line);
        else if (item.action === "submit_reject_f") saveRejectZoneF(item.data, item.user, item.line);
        else if (item.action === "submit_downtime_c") saveDowntimeZoneC(item.data, item.user, item.line);
        else if (item.action === "submit_downtime_f") saveDowntimeZoneF(item.data, item.user, item.line);
        else if (item.action === "update_reject_c") updateRejectZoneC(item.data, item.user, item.line);
        else if (item.action === "update_reject_f") updateRejectZoneF(item.data, item.user, item.line);
        else if (item.action === "update_downtime_c") updateDowntimeZoneC(item.data, item.user, item.line);
        else if (item.action === "update_downtime_f") updateDowntimeZoneF(item.data, item.user, item.line);
      } catch (err) {
        item.retries = (item.retries || 0) + 1;
        if (item.retries < 5) remainingQueue.push(item);
      }
    }
    props.setProperty("PENDING_SAVES", JSON.stringify(remainingQueue));
  } catch (e) { console.error("Error processPendingSaves:", e); }
}

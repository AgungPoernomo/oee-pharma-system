DROP TABLE IF EXISTS downtime_line3_zonec;
CREATE TABLE downtime_line3_zonec (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tanggal DATE,
    shift VARCHAR(10),
    `group` VARCHAR(10),
    no_batch VARCHAR(50),
    start_h INT DEFAULT 0,
    start_m INT DEFAULT 0,
    end_h INT DEFAULT 0,
    end_m INT DEFAULT 0,
    duration INT DEFAULT 0,
    plan_unplan VARCHAR(20),
    root_cause VARCHAR(50),
    proses VARCHAR(50),
    unit VARCHAR(100),
    kasus TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_dt_c_tanggal (tanggal)
